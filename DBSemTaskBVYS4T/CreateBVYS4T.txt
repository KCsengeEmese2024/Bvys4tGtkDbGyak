create table Csapatok (
 Csapat_id char(10) primary key,
 Csapat_név char(40),
 Származás char(15),
 Székhely char(15),
 Csapatfőnök char(16)
);

create table Pilóták (
 Rajtszám int(2) primary key,
 Név char(20),
 Származás char(10), 
 Magasság_cm int(3),
 Kor int(2) check (Kor>0),
 Konstruktőr char(10),
 Foreign key (Konstruktőr) references Csapatok(Csapat_id)
);

create table Nagydíjak (
 Verseny_id int(2) primary key,
 Nagydíj char(25),
 Kontinens char(15),
 Ország char(25),
 Város char(15),
 Dátum date,
 Körök_száma int(2),
 Pálya_hossza_méterben int(4)
);

create table Autó_modell (
 Autó_id char(5) primary key,
 Motor char(20),
 Üzemanyag char(25),
 Csapat char(10),
 Foreign key (Csapat) references Csapatok(Csapat_id)
);

create table Eredmény (
 Eredmény_id int(2) primary key,
 Nagydíjon int(2), Foreign key (Nagydíjon) references Nagydíjak(Verseny_id),
 Versenyző_rajtszáma int(2), Foreign key (Versenyző_rajtszáma) references Pilóták(Rajtszám),
 Pozíció int(2),
 Pontszám int(2)
);
